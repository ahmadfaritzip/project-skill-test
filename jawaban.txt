class Worker {
  constructor(hoursWorked, rate) {
    this.hoursWorked = hoursWorked;
    this.rate = rate;
    this.TAX = 0.15; // misalnya nilai pajak tetap adalah 15%
  }
  
  basicSalary() {
    return this.hoursWorked * this.rate;
  }

  overviewSalary() {
    return this.basicSalary() - this.TAX;
  }
}
